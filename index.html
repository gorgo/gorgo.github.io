varying vec2 vN; void main() { vec4 p = vec4( position, 1. ); vec3 e = normalize( vec3( modelViewMatrix * p ) ); vec3 n = normalize( normalMatrix * normal ); vec3 r = reflect( e, n ); float m = 2. * sqrt( pow( r.x, 2. ) + pow( r.y, 2. ) + pow( r.z + 1., 2. ) ); vN = r.xy / m + .5; gl_Position = projectionMatrix * modelViewMatrix * p; }
uniform sampler2D tMatCap; varying vec2 vN; void main() { vec3 base = texture2D( tMatCap, vN ).rgb; gl_FragColor = vec4( base * 1.5, 1. ); }
varying vec2 vNr; varying vec2 vNg; varying vec2 vNb; uniform float ratio; uniform float normalMul; void main() { vec4 p = vec4( position, 1. ); vec3 e = normalize( vec3( modelViewMatrix * p ) ); vec3 n = normalize( normalMatrix * normal ); const float DELTA = 0.002; vec3 r; float m; r = refract( e, n, ratio); m = 2. * sqrt( pow( r.x, 2. ) + pow( r.y, 2. ) + pow( r.z + 1., 2. ) ); vNr = r.xy + .5; r = refract( e, n, ratio + DELTA); m = 2. * sqrt( pow( r.x, 2. ) + pow( r.y, 2. ) + pow( r.z + 1., 2. ) ); vNg = r.xy + .5; r = refract( e, n, ratio + DELTA + DELTA ); m = 2. * sqrt( pow( r.x, 2. ) + pow( r.y, 2. ) + pow( r.z + 1., 2. ) ); vNb = r.xy + .5; gl_Position = projectionMatrix * modelViewMatrix * p; }
uniform sampler2D tMatCap; varying vec2 vNr; varying vec2 vNg; varying vec2 vNb; void main() { vec3 base; base.r = texture2D( tMatCap, vNr ).r; base.g = texture2D( tMatCap, vNg ).g; base.b = texture2D( tMatCap, vNb ).b; gl_FragColor = vec4( base * 1.0 + 0.05, 1. ); }
